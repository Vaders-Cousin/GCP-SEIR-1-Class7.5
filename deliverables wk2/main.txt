resource "google_compute_firewall" "vaderaccess_allow_http" {
  name    = "vaderaccess-allow-http"
  network = google_compute_network.vpc_network.name 

  allow {
    protocol = "tcp"
    ports    = ["80"]
  }

  source_ranges = ["0.0.0.0/0"]

  # target_tags = ["we-got-this-ish-lab2web"]
}

#Chewbacca: The compute instance—your first reproducible node.
resource "google_compute_instance" "we-got-this-ish-lab2" {
  name         = var.vm_name
  machine_type = "e2-micro"
  zone         = var.zone

  boot_disk {
    initialize_params {
      image = "debian-cloud/debian-12"
    }
  }

  network_interface {
    subnetwork = google_compute_subnetwork.vaderland.name

    access_config {} # External IP
  }

  metadata = {
    #Chewbacca: The banner is identity. Make it yours.
    student_name = var.student_name
  }

  metadata_startup_script = file("./startup_script.sh")

  # tags = ["we-got-this-ish-lab2web"]
}

#Chewbacca: Outputs are how automation speaks to other automation.
output "vm_external_ip" {
  value = google_compute_instance.we-got-this-ish-lab2.network_interface[0].access_config[0].nat_ip
}

output "vm_url" {
  value = "http://${google_compute_instance.we-got-this-ish-lab2.network_interface[0].access_config[0].nat_ip}"
}
